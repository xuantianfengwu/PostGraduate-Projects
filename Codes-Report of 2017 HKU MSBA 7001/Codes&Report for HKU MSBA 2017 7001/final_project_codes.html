def origin_data_clean(in_data,years=10):
    data = in_data.copy()
    print('start origin data cleaning! It will take around 5 minutes.')
    data.columns = ['symbol','date','time','open','high','low','close','vol']
    del data['vol']
    data['time'] = data['time'].apply(lambda x:str(x).zfill(6)[:-2])
    data['datetime'] = [datetime.datetime.strptime(str(x)+y,'%Y%m%d%H%M') for x,y in zip(data['date'],data['time'])]
    data['wkd'] = data['datetime'].apply(lambda x:x.weekday()+1)
    begind = data['date'].tolist()[-1] - years * 10000
    data = data[data['date']>=begind].sort_values('datetime')
    first_wkd7 = data[data['wkd']==7]['date'].tolist()[0]
    data = data[data['date']>=first_wkd7].reset_index(drop=True)
    last_wkd5 = data[data['wkd']==5]['date'].tolist()[-1]
    data = data[data['date']<=last_wkd5].reset_index(drop=True)
    data['date'] = data['date'].apply(lambda x:str(x))
    data['day'] = data['date'].apply(lambda x:x[-2:])
    data['month'] = data['date'].apply(lambda x:x[-4:-2])
    data['hour'] = data['time'].apply(lambda x:x[:2])
    data['minute'] = data['time'].apply(lambda x:x[2:])
    weeks = get_weeks(data['date'].tolist())
    print('origin data cleaning finished! Shape = {}'.format(data.shape))
    return data,weeks
def get_weeks(days):
    first_d1 = datetime.datetime.strptime(str(days[0]),'%Y%m%d')
    last_d6 = datetime.datetime.strptime(str(days[-1]),'%Y%m%d')
    first_d6 = first_d1+datetime.timedelta(days=5)
    now_week = (first_d1,first_d6)
    weeks = []
    while now_week[-1]<=last_d6:
        weeks.append(now_week)
        now_week = (now_week[0]+datetime.timedelta(days=7),now_week[1]+datetime.timedelta(days=7))
    weeks = [(str(x).replace('-','')[:8],str(y).replace('-','')[:8]) for x,y in weeks]
    return weeks

For each week, do the following operations¶

1. Cut the beginning to a closest time whose minute is (0 or 30), Cut the ending to a closest time whose minute is (29 or 59). Thus the final mktdata¶

2. Add every missing minute in the middle (we add regardless of any market closing. We think this kind of adding way is appropriate because Forex Market is a global, 24hr market, so when any country has its own holiday and close the market, other countries' markets are still opening because they don't have the same holiday.¶

Then the mktdata will become complete¶

def add_lacked_data(input_df):
    df = input_df.copy()
    df.sort_values('datetime',inplace=True);df.drop_duplicates(inplace=True);df.reset_index(drop=True,inplace=True)
    datetimes = df['datetime'].tolist();cols = df.columns
    first_time,last_time = datetimes[0],datetimes[-1]
    #cut/add beginning
    if first_time.minute not in [0,30]:
        nums = first_time.minute
        if nums==0:
            df = df.drop(df.index.tolist()[0])
        elif 1<=nums<=15: #minute<15to 0
            example = df.iloc[0].tolist()
            to_append = [example.copy() for i in range(nums)]
            for i in range(nums):
                to_append[i][7] = first_time - datetime.timedelta(minutes=i+1)
            df = df.append(pd.DataFrame(to_append,columns=cols))
        elif 15<nums<30:#minute in(16,29)add to 30
            target_datetime = datetime.datetime(first_time.year,first_time.month,first_time.day,first_time.hour,30,0)
            temp = df[(df['datetime']>=first_time)&(df['datetime']<target_datetime)]
            if len(temp)>0:
                df = df.drop(temp.index.tolist())
        elif 30<nums<=45: #minute in (30,45)to 30
            example = df.iloc[0].tolist()
            to_append = [example.copy() for i in range(nums-30)]
            for i in range(nums-30):
                to_append[i][7] = first_time - datetime.timedelta(minutes=i+1) #datetime
                open_price = example[3]
                to_append[i][4],to_append[i][5],to_append[i][6] = open_price,open_price,open_price
            df = df.append(pd.DataFrame(to_append,columns=cols))
        else: #minute in (45,59]to 0 of next minute
            target_datetime = first_time + datetime.timedelta(minutes = 60-nums)
            temp = df[(df['datetime']>=first_time)&(df['datetime']<target_datetime)]
            if len(temp)>0:
                df = df.drop(temp.index.tolist())
    df.sort_values('datetime',inplace=True);df.reset_index(drop=True,inplace=True)
    #cut/add ending
    if last_time.minute not in [29,59]:
        nums = last_time.minute
        if nums<29: #minute in 028],to last minute's 59
            target_datetime = last_time - datetime.timedelta(minutes = nums+1)
            temp = df[(df['datetime']>target_datetime)&(df['datetime']<=last_time)]
            if len(temp)>0:
                df = df.drop(temp.index.tolist())
        elif 29<nums<59: # minute in [30-58]to 29
            target_datetime = datetime.datetime(last_time.year,last_time.month,last_time.day,last_time.hour,29,0)
            temp = df[(df['datetime']>target_datetime)&(df['datetime']<=last_time)]
            if len(temp)>0:
                df = df.drop(temp.index.tolist())
    # resort datetime,reset_index,then add missing data in the middle       
    df.reset_index(drop=True,inplace=True)
    datetimes = df['datetime'].tolist()
    lack_bars = ((datetimes[-1] - datetimes[0]).days*24*60 +(datetimes[-1] - datetimes[0]).seconds/60)+1 - len(df)
    if lack_bars>0: # Lack data in the middle
        for ii,t1,t2 in zip(df.index[1:],datetimes[:-1],datetimes[1:]):
            diff_min = (t2-t1).seconds/60
            if diff_min>1:
                nums = int(diff_min) -1
                example = df.iloc[ii].tolist()
                to_append = [example.copy() for i in range(nums)]
                for i in range(nums):
                    to_append[i][7] = example[7] - datetime.timedelta(minutes=i+1)
                    open_price = example[3]
                    to_append[i][4],to_append[i][5],to_append[i][6] = open_price,open_price,open_price
                df = df.append(pd.DataFrame(to_append,columns=cols))
                lack_bars-=nums
                if lack_bars==0:
                    break
    df.sort_values('datetime',inplace=True);df.reset_index(drop=True,inplace=True)
    # remake columns of "date" and "time"
    df['date'] = df['datetime'].apply(lambda x:str(x)[:10].replace('-',''))
    df['time'] = df['datetime'].apply(lambda x:str(x)[11:16].replace(':',''))
    return df

def make_data_clean(data,weeks):
    print('data completeness check begin! The data shape is {}'.format(data.shape))

def aggregate_min_bar(input_df,freq=240):
    df = input_df.copy().reset_index(drop=True)
    final = []
    if len(df)%freq==0:
        start_points = sorted(range(len(df)))[::freq]
    else:
        print('df_shape={},freq={},(len(df)/freq) is not an integer'.format(df.shape,freq))
        print('will drop the last {} bar to aggregate successfully\n'.format(len(df)%freq))
        start_points = sorted(range(len(df)))[::freq][:-1]
    k,total,start=0,len(start_points),time.time()
    print('start aggregating!')
    for i in start_points:
        k+=1;view_perc_bar(k,total,start)
        temp = df[i:i+freq];first_line = temp.iloc[0,:].tolist()
        first_line[4],first_line[5],first_line[6] = temp['high'].max(),temp['low'].min(),temp['close'].tolist()[-1]
        final.append(first_line)
    new_df =pd.DataFrame(final)
    if len(final)>0:
        new_df.columns = input_df.columns
    print('\naggregating finished! Now the data shape is {}'.format(new_df.shape))
    new_df['r'] = (new_df['close']/new_df['close'].shift(1))-1
    new_df['lag_r'] = new_df['r'].shift(1)
    new_df['next_r'] = new_df['r'].shift(-1)
    new_df['lag_r_ma10'] = pd.rolling_mean(new_df['r'],10)
    new_df.dropna(inplace=True); new_df.reset_index(drop=True,inplace=True)
    return new_df

def get_new_weeks(weeks):
    new_weeks = []
    for w in weeks:
        d1,d2 = w
        new_w = get_range_days(d1,d2)
        new_weeks.append(new_w)
    return new_weeks

def get_range_days(begind,endd):
    dates = []  
    d1 = datetime.datetime.strptime(begind, "%Y%m%d")  
    d2 = datetime.datetime.strptime(endd, "%Y%m%d")  
    date = d1 
    while date <= d2:
        dates.append(date)
        date = date+datetime.timedelta(days=1)
    dates = [str(d)[:10].replace('-','') for d in dates]
    return dates  

def get_daily_cftc_data(n,in_date,total,start):
    view_perc_bar(n+1,total,start)
    date = str(in_date)[:4]+'-'+str(in_date)[4:6]+'-'+str(in_date)[6:]
    page = 'http://service.99qh.com/cftc/GetData.ashx?code=JAPANESE%20YEN%20-%20CHICAGO%20MERCANTILE%20EXCHANGE&date={}'.format(date)
    get = False
    num = 1
    while get==False or num>20:
        try:
            req = requests.get(page,timeout=10)
            get = True
        except:
            sys.stdout.write('\r')
            print('Time out for {}. Time out times: {}. Will try again!'.format(in_date,num))
            num+=1
    if num>20:
        sys.stdout.write('\r')
        print("Have tried and failed so many times for the day's data, so we give up this day.")
    soup = BeautifulSoup(req.text,'lxml')
    contents = [x.replace(',','') for x in soup.get_text().split('@')]
    df = contents[:10]+contents[30:50]+contents[-29:-21]
    df = pd.DataFrame([in_date]+df).T
    df.columns = ['until_date','total','fund_l','fund_s','fund_hedge','merch_l','merch_s','reported_l','reported_s','unrep_l','unrep_s',

def get_cftc_data(weeks):
    complete_weeks = get_new_weeks(weeks)
    cftc_days = [x[2] for x in complete_weeks]
    n,total,start = 0,len(cftc_days),time.time()
    print('Start scrapping cftc_data! Totally {} days to scrap'.format(total))
    cftc_dfs = [get_daily_cftc_data(n,d,total,start) for n,d in enumerate(cftc_days)]
    cftc_final = pd.concat(cftc_dfs)
    print('\ncftc_data finished! data shape is {}'.format(cftc_final.shape))
    return cftc_final

def get_index_data(input_df):
    print('idx_data scrapping start!')
    dates = input_df['date'].tolist()
    start =  dates[0]
    end = dates[-1]
    sp500 = web.DataReader("^GSPC", 'yahoo',start, end)
    nl225 =  web.DataReader("^N225", 'yahoo', start, end)
    sp500.reset_index(inplace=True)
    nl225.reset_index(inplace=True)
    nl = nl225[['Date']]
    sp = sp500[['Date']]
    nl['return'] = nl225['Adj Close']/nl225['Adj Close'].shift(1)-1
    sp['return'] = sp500['Adj Close']/sp500['Adj Close'].shift(1)-1
    nl.columns = ['date','nl_r']
    sp.columns = ['date','sp_r']
    dff = pd.merge(nl,sp,how='outer',on='date')
    dff = dff[1:].fillna(0).reset_index(drop=True)
    dff['date'] = dff['date'].apply(lambda x:str(x)[:10].replace('-',''))
    print('idx_data scrapping finished! data length is {}'.format(len(dff)))
    return dff

def get_indicators(new_df):
    print('Start calculating indicators')
    O = np.array(new_df['open'])
    H = np.array(new_df['high'])
    L = np.array(new_df['low'])
    C = np.array(new_df['close'])
    indi_df = new_df[['datetime','close','open','high','low']]
    indi_df['adx'] = talib.ADX(H,L,C,14)
    indi_df['sma'] = talib.SMA(C,10)
    indi_df['ema'] = talib.EMA(C,10)
    indi_df['sar'] = talib.SAR(H,L)
    indi_df['stoch_k'],indi_df['stoch_d'] = talib.STOCH(H,L,C)
    indi_df['atr'] = talib.ATR(H,L,C)
    indi_df['std'] = talib.STDDEV(C)
    indi_df['willr'] = talib.WILLR(H,L,C)
    indi_df['stochrsi_k'],indi_df['stochrsi_d'] = talib.STOCHRSI(C)
    indi_df['rsi'] = talib.RSI(C)
    indi_df['typprice'] = talib.TYPPRICE(H,L,C)
    indi_df = indi_df.dropna().reset_index(drop=True)
    print('indi_data calculating finished! length is {}'.format(len(indi_df)))
    return indi_df

V.Then we merge the indi_data,cftc_data,mkt_data,idx_data together to get our final_df.¶

It really takes much time to finish this process because we try our best to avoid "future data". For example:¶

1.for cftc_df, the original date is "until date", not "publish date", the "publish date" is on Saturday, and is 4 day later then the "until date". So we merge cftc_data with the mktdata the next week.¶

2.For idx_data, because S&P 500 and Nikkei 225 only can traded for several hours every day. And the timezones of their markets (one is GMT-5, one is GMT+7) are different from Forex data, which is GMT+0. So we have to merge them to our mkt data with a shift. This process takes most of the merging time when you run the codes below.¶

def combine_cftc_and_mkdata(cftc_final,mkdata,weeks=weeks):
    cftc_final.reset_index(drop=True,inplace=True)
    final = []
    for i,w in zip(range(len(cftc_final)-1),weeks[1:]): 
        d1,d2= w
        cftc_temp = cftc_final.iloc[[i],:]
        cftc = pd.concat([cftc_temp]*6)
        days = get_range_days(d1,d2)
        cftc['until_date'] = days
        final.append(cftc)
    cftc_df = pd.concat(final)
    cftc_df.columns = ['date']+cftc_final.columns.tolist()[1:]
    final = pd.merge(mkdata,cftc_df,how='left',on='date').dropna()
    print('cftc merge finished!')
    return final

VII. Lbel, train and Test:¶

we change the "thresh" parameter of the function "add_label" little by little and record the test performance of our models which is trained on train and tested on test.¶

1.Add label according to the relationship of thresh and next_r(the return of the next bar)¶

label = 1 <------- next_r > thresh¶

label = 0 <------- -thresh < next_r < thresh¶

label = -1 <------- next_r < -thresh¶

2.Split data into train(first 80%) and test(last 20%)¶

3. Train models with the train data and test on the test data¶

def to_label(i,thresh=0):
    if i<0-thresh:
        l = -1
    elif i>0+thresh:
        l = 1
    else:
        l = 0
    return l

def add_label(input_df,thresh=0):
    df = input_df.copy()
    df['label'] = df['next_r'].apply(lambda x:to_label(x,thresh))
    return df

def split_data(input_df,perc=0.8):
    df = input_df.copy()
    df.reset_index(drop=True,inplace=True)
    length = len(df);  train_length = int(length*perc)
    train,test= df[:train_length],df[train_length:]
    train_X = train[[i for i in train.columns if i not in ['next_r','label']]]
    test_X = test[[i for i in train.columns if i not in ['next_r','label']]]
    train_Y = train[['next_r','label']]
    test_Y = test[['next_r','label']]
    return train_X,train_Y,test_X,test_Y

def max_drawdown(in_returns):
    returns=[1]
    for i in in_returns:
        returns.append(returns[-1]+i)
    max_returns = np.maximum.accumulate(returns)
    draw = returns / max_returns
    max_draw = np.minimum.accumulate(draw)
    draw_series = (1 - max_draw)
    return draw_series

def get_test_results(model,train_X,train_Y,test_X,test_Y):
    result = model.fit(train_X,train_Y[['label']])
    test_Y['pred_label'] = result.predict(test_X)
    up_df = test_Y[test_Y['label']==1]
    down_df = test_Y[test_Y['label']==-1]
    buy_df = test_Y[test_Y['pred_label']==1]
    sell_df = test_Y[test_Y['pred_label']==-1]
    total = len(test_Y); long_total = len(buy_df); short_total = len(sell_df); up_total = len(up_df); down_total = len(down_df)
    chances = (len(up_df)+len(down_df));trades = (len(buy_df)+len(sell_df))
    right_pred = len(test_Y[test_Y['pred_label']==test_Y['label']]); 
    right_up = len(up_df[up_df['pred_label']==up_df['label']]); right_down = len(down_df[down_df['pred_label']==down_df['label']])
    right_long = len(buy_df[buy_df['pred_label']==buy_df['label']]); right_short = len(sell_df[sell_df['pred_label']==sell_df['label']])
    long_return = buy_df['next_r'].sum(); short_return = ((-1)*sell_df['next_r']).sum()
    results = [ chances/total , trades , right_pred/total, right_up/up_total if up_total>0 else 0,right_down/down_total if down_total>0 else 0,
                 right_long/long_total if long_total>0 else 0, right_short/short_total if short_total>0 else 0, 
                (long_return+short_return)/(long_total+short_total)*100 if (long_total+short_total)>0 else 0, 
                 long_return/long_total*100 if long_total>0 else 0, short_return/short_total*100 if short_total>0 else 0,
                 max_drawdown(test_Y['next_r']*test_Y['pred_label']).max()*100,(test_Y['next_r']*test_Y['pred_label']).sum()*100]
    results = [round(x,3) for x in results]
    weights = result.feature_importances_ if type(model)==RandomForestClassifier else abs(result.coef_).sum(axis=0)
    return results,weights

def get_result_panel(df,model,threshes):
    items = ['Not_0_Perc','Trades','Accuracy','UpAccu','DownAccu','LongAccu','ShortAccu','AvgR(%)/trade','AvgR(%)/Long','AvgR(%)/Short','MAXDD(%)','Cum_R(%)']
    results = []; t_weights = []
    n,total,start = 0,len(threshes),time.time()
    for th in threshes:
        ddf= add_label(df,th)
        train_X,train_Y,test_X,test_Y = split_data(ddf)
        result,weights = get_test_results(model,train_X,train_Y,test_X,test_Y)
        n+=1; view_perc_bar(n,total,start)
        t_weights.append(weights)
        results.append(result)
    sys.stdout.write('\r'); sys.stdout.flush()
    weight_df = pd.DataFrame(t_weights,columns=train_X.columns)
    weight_final = pd.concat([weight_df.mean(),weight_df.std(),weight_df.mean()/weight_df.std()],axis=1).fillna(0)
    weight_final.columns=['mean','std','R']
    final = pd.DataFrame(results,columns = items)
    final['threshes'] = threshes; final.set_index('threshes',inplace=True)
    result = {str(type(model)):final}
    return result,weight_final
